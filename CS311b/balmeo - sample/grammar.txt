<stmt> -> <assign>
<stmt> -> <declaration>

<expr1> -> <expr2> <expr1'>
<expr1'> -> <plus> <expr2> <expr1'>
<expr1'> -> EPSILON
<expr2> -> <term> <expr2'>
<expr2'> -> <times> <term> <expr2'>
<expr2'> -> EPSILON
<declaration> -> <var> <iden> <colon> <type> <initial_value> <semi>
<initial_value> -> <eq> <term>
<initial_value> -> EPSILON
<term> -> <iden>
<term> -> <const>
